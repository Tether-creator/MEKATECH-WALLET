async function fetchBalance() { try { let response = await fetch("https://mekatech-wallet.onrender.com/balance", { method: "GET", headers: { "Content-Type": "application/json" } }); if (!response.ok) { throw new Error(HTTP error! Status: ${response.status}); } let data = await response.json(); document.querySelector(".balance").textContent = Balance: ${data.Balance}; } catch (error) { document.querySelector(".balance").style.display = "none"; let errorElement = document.querySelector(".error"); errorElement.style.display = "block"; errorElement.textContent = "Error fetching balance. Please try again later."; console.error("Fetch Error:", error); } } fetchBalance();
